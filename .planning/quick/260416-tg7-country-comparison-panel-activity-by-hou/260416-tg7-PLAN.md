1. `load_activity_by_hour_country(engine, top_n: int = 3) -> pd.DataFrame`
   Use a CTE for `top_countries`, then assign each session its modal `conn_country`
   via `mode() WITHIN GROUP (ORDER BY rp.conn_country)` joined on
   `rp.played_at BETWEEN s.start_time AND s.end_time`. Use `s.hour_of_day` (already
   local-adjusted in build_sessions) — do NOT re-derive from played_at.
   Final SELECT:
   ```sql
   WITH top_countries AS (
       SELECT conn_country
       FROM raw_plays
       WHERE conn_country IS NOT NULL
       GROUP BY conn_country
       ORDER BY COUNT(*) DESC
       LIMIT :top_n
   ),
   session_country AS (
       SELECT s.session_id,
              s.hour_of_day,
              mode() WITHIN GROUP (ORDER BY rp.conn_country) AS conn_country
       FROM sessions s
       JOIN raw_plays rp
         ON rp.played_at BETWEEN s.start_time AND s.end_time
       WHERE rp.conn_country IS NOT NULL
       GROUP BY s.session_id, s.hour_of_day
   )
   SELECT sc.conn_country         AS country_code,
          sc.hour_of_day           AS hour,
          COALESCE(al.activity_label, 'unlabeled') AS activity_label,
          COUNT(*)                 AS sessions
   FROM session_country sc
   LEFT JOIN activity_labels al ON al.session_id = sc.session_id
   WHERE sc.conn_country IN (SELECT conn_country FROM top_countries)
   GROUP BY sc.conn_country, sc.hour_of_day, activity_label
   ORDER BY sc.conn_country, sc.hour_of_day
   ```
   Do NOT add country_name in SQL — let app.py map via `_COUNTRY_NAMES` (mirrors
   the pattern used by the Global Footprint section). Column list returned by
   this function is [country_code, hour, activity_label, sessions]; the
   must_haves list `country_name` because app.py enriches before plotting.

2. `load_top_tracks_by_country(engine, top_n: int = 3, limit: int = 5) -> pd.DataFrame`
   ```sql
   WITH top_countries AS (
       SELECT conn_country
       FROM raw_plays
       WHERE conn_country IS NOT NULL
       GROUP BY conn_country
       ORDER BY COUNT(*) DESC
       LIMIT :top_n
   ),
   per_country AS (
       SELECT conn_country AS country_code,
              track_name,
              artist_name,
              COUNT(*) AS plays,
              ROW_NUMBER() OVER (
                  PARTITION BY conn_country
                  ORDER BY COUNT(*) DESC
              ) AS rn
       FROM raw_plays
       WHERE conn_country IN (SELECT conn_country FROM top_countries)
       GROUP BY conn_country, track_name, artist_name
   )
   SELECT country_code, track_name, artist_name, plays
   FROM per_country
   WHERE rn <= :limit
   ORDER BY country_code, plays DESC
   ```
   Params: `{"top_n": top_n, "limit": limit}`.

3. `load_country_stats(engine, top_n: int = 3) -> pd.DataFrame`
   ```sql
   WITH top_countries AS (
       SELECT conn_country
       FROM raw_plays
       WHERE conn_country IS NOT NULL
       GROUP BY conn_country
       ORDER BY COUNT(*) DESC
       LIMIT :top_n
   ),
   plays_agg AS (
       SELECT conn_country AS country_code,
              COUNT(*) AS plays,
              ROUND(SUM(duration_ms) / 60000.0, 1) AS minutes_played
       FROM raw_plays
       WHERE conn_country IN (SELECT conn_country FROM top_countries)
       GROUP BY conn_country
   ),
   session_country AS (
       SELECT s.session_id,
              s.hour_of_day,
              s.duration_minutes,
              mode() WITHIN GROUP (ORDER BY rp.conn_country) AS conn_country
       FROM sessions s
       JOIN raw_plays rp
         ON rp.played_at BETWEEN s.start_time AND s.end_time
       WHERE rp.conn_country IS NOT NULL
       GROUP BY s.session_id, s.hour_of_day, s.duration_minutes
   ),
   session_agg AS (
       SELECT conn_country AS country_code,
              ROUND(AVG(duration_minutes)::numeric, 1) AS avg_session_duration
       FROM session_country
       WHERE conn_country IN (SELECT conn_country FROM top_countries)
       GROUP BY conn_country
   ),
   dominant AS (
       SELECT country_code, activity_label AS dominant_activity
       FROM (
           SELECT sc.conn_country AS country_code,
                  al.activity_label,
                  COUNT(*) AS cnt,
                  ROW_NUMBER() OVER (
                      PARTITION BY sc.conn_country
                      ORDER BY COUNT(*) DESC
                  ) AS rn
           FROM session_country sc
           JOIN activity_labels al ON al.session_id = sc.session_id
           WHERE al.activity_label != 'unknown'
             AND sc.conn_country IN (SELECT conn_country FROM top_countries)
           GROUP BY sc.conn_country, al.activity_label
       ) ranked
       WHERE rn = 1
   ),
   peak AS (
       SELECT country_code, hour AS top_hour
       FROM (
           SELECT sc.conn_country AS country_code,
                  sc.hour_of_day AS hour,
                  COUNT(*) AS cnt,
                  ROW_NUMBER() OVER (
                      PARTITION BY sc.conn_country
                      ORDER BY COUNT(*) DESC
                  ) AS rn
           FROM session_country sc
           WHERE sc.conn_country IN (SELECT conn_country FROM top_countries)
           GROUP BY sc.conn_country, sc.hour_of_day
       ) ranked
       WHERE rn = 1
   )
   SELECT pa.country_code,
          pa.plays,
          pa.minutes_played,
          sa.avg_session_duration,
          d.dominant_activity,
          p.top_hour
   FROM plays_agg pa
   LEFT JOIN session_agg sa ON sa.country_code = pa.country_code
   LEFT JOIN dominant    d  ON d.country_code  = pa.country_code
   LEFT JOIN peak        p  ON p.country_code  = pa.country_code
   ORDER BY pa.plays DESC
   ```
   Use `df["dominant_activity"] = df["dominant_activity"].fillna("unlabeled")`
   before returning.

Follow the existing docstring + SQL-in-triple-quotes style. Do NOT refactor
